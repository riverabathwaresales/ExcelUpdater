VERSION 1.0 CLASS
BEGIN
  MultiUse = -1  'True
END
Attribute VB_Name = "ThisWorkbook"
Attribute VB_GlobalNameSpace = False
Attribute VB_Creatable = False
Attribute VB_PredeclaredId = True
Attribute VB_Exposed = True
Option Explicit

Public isLoggedIn As Boolean

Private Sub Workbook_Open()

    On Error GoTo safeFail

    Dim savedUser As String

    If Not HasSessionCache() Then
        ClearSavedLogin
    End If

    savedUser = Trim(GetSetting(APP_NAME, "Login", "User", ""))

    If savedUser <> "" Then
        isLoggedIn = True

        If ApplyDailyUserValidation(savedUser) Then
            EnableVBAFeatures
        Else
            isLoggedIn = False
            DisableVBAFeatures
        End If

        Exit Sub
    End If

    DisableVBAFeatures

    If PerformLogin() Then
        isLoggedIn = True
    Else
        isLoggedIn = False
    End If

    Exit Sub

safeFail:
    isLoggedIn = False
    DisableVBAFeatures

End Sub

Private Sub Workbook_SheetChange(ByVal sh As Object, ByVal Target As Range)

    On Error GoTo safeExit

    If isLoggedIn = False Then Exit Sub
    If vbaFeaturesEnabled = False Then Exit Sub
    If isRebuildingSummary Then Exit Sub

    Application.EnableEvents = False
    Application.ScreenUpdating = False

    If IsBathSheet(sh) Then

        Dim cell As Range
        Dim needSummaryRefresh As Boolean

        For Each cell In Target.Cells

            If IsProtectedTotalRow(sh, cell.Row) Then GoTo nextCell

            If cell.Column = 1 Then
                If Trim(cell.Value) = "" Then
                    ClearRow cell
                Else
                    FillDataAndImage cell
                End If

                needSummaryRefresh = True
            End If

            If cell.Column = 5 Then
                RecalculateRow sh, cell.Row
                needSummaryRefresh = True
            End If

            If cell.Column = 4 Or cell.Column = 12 Or cell.Column = 13 Or cell.Column = 14 Then
                needSummaryRefresh = True
            End If

nextCell:
        Next cell

        If NeedAdvancedSummaryRefresh(sh, Target) Then
            needSummaryRefresh = True
        End If

        If needSummaryRefresh Then
            sh.Calculate

            RebuildSummaryFast
            RebuildSummaryAdvanced

            If Not ActiveSheet Is Nothing Then
                If IsSummarySheet(ActiveSheet) Then
                    ActiveSheet.Calculate
                End If
            End If
        End If

    ElseIf IsSummarySheet(sh) Then

        HandleSummarySheetChange sh, Target
        ApplyTotalRowFormula sh
        sh.Calculate

    End If

safeExit:
    Application.ScreenUpdating = True
    Application.EnableEvents = True

End Sub

Private Sub Workbook_SheetCalculate(ByVal sh As Object)

    On Error GoTo safeExit

    If isLoggedIn = False Then Exit Sub
    If vbaFeaturesEnabled = False Then Exit Sub
    If isRebuildingSummary Then Exit Sub

    If IsSummarySheet(ActiveSheet) Then
        If IsBathSheet(sh) Then
            Application.EnableEvents = False
            Application.ScreenUpdating = False

            RebuildSummaryFast
            RebuildSummaryAdvanced
            ActiveSheet.Calculate
        End If
    End If

safeExit:
    Application.ScreenUpdating = True
    Application.EnableEvents = True

End Sub

Private Sub Workbook_SheetActivate(ByVal sh As Object)

    On Error GoTo safeExit

    If vbaFeaturesEnabled = False Then Exit Sub

    If IsSummarySheet(sh) Then
        RebuildSummaryFast
        RebuildSummaryAdvanced
    End If

safeExit:

End Sub

Private Sub Workbook_BeforeClose(Cancel As Boolean)

    On Error Resume Next

    Application.CutCopyMode = False

    If Not imgWB Is Nothing Then
        imgWB.Saved = True
    End If

    Set imgWB = Nothing

End Sub

Private Sub Workbook_NewSheet(ByVal sh As Object)

    On Error GoTo safeExit

    If vbaFeaturesEnabled = False Then Exit Sub
    If Not IsBathSheet(sh) Then Exit Sub

    Application.EnableEvents = False
    Application.ScreenUpdating = False

    RebuildSummaryFast
    RebuildSummaryAdvanced

safeExit:
    Application.ScreenUpdating = True
    Application.EnableEvents = True

End Sub

Private Sub Workbook_SheetBeforeDelete(ByVal sh As Object)

    On Error Resume Next

    If vbaFeaturesEnabled = False Then Exit Sub
    If Not IsBathSheet(sh) Then Exit Sub

    ScheduleSummaryRefreshAfterStructureChange

End Sub
